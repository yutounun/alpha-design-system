The **Drawer** is a bottom (or top) sheet built on **Vaul**. For side panels, use **Sheet** instead. `direction` accepts `top` or `bottom` (`bottom` is the default).

## Usage

```tsx
import { Button } from "@/components/button"
import {
    Drawer,
    DrawerClose,
    DrawerContent,
    DrawerDescription,
    DrawerFooter,
    DrawerHeader,
    DrawerTitle,
    DrawerTrigger,
} from "@/components/drawer"

export function Example() {
    return (
        <Drawer direction="bottom">
            <DrawerTrigger asChild>
                <Button variant="outline">Open</Button>
            </DrawerTrigger>
            <DrawerContent>
                <DrawerHeader>
                    <DrawerTitle>Title</DrawerTitle>
                    <DrawerDescription>Description.</DrawerDescription>
                </DrawerHeader>
                <DrawerFooter>
                    <DrawerClose asChild>
                        <Button variant="outline">Close</Button>
                    </DrawerClose>
                </DrawerFooter>
            </DrawerContent>
        </Drawer>
    )
}
```
